import random

# 100 words, all >= 5 letters
word_list = [
    "python", "random", "school", "planet", "object", "method", "import", "string", "circle", "matrix",
    "pickle", "button", "window", "canvas", "scroll", "layout", "binary", "doctor", "engine", "guitar",
    "jungle", "legend", "memory", "option", "policy", "reason", "signal", "thread", "update", "viewer",
    "yellow", "zombie", "access", "buffer", "center", "driver", "effect", "filter", "global", "hidden",
    "impact", "jargon", "kernel", "layout", "mirror", "native", "object", "packet", "quote",  "repair",
    "socket", "target", "upload", "visual", "widget", "xenial", "yearly", "zapper", "abroad", "belief",
    "climax", "define", "effort", "format", "garage", "honest", "intent", "jockey", "kidney", "laptop",
    "manage", "notice", "output", "pirate", "quirky", "rescue", "sprint", "tunnel", "unique", "vaccum",
    "wealth", "xrayman", "yonder", "zenith", "anchor", "border", "charge", "damage", "effect", "fossil",
    "glider", "hammer", "island", "jacket", "kitten", "launch", "modern", "number", "oracle", "parrot"
]

while True:
    current_word = random.choice(word_list)
    hidden = ["_"] * len(current_word)
    hidden[0] = current_word[0]
    hidden[-1] = current_word[-1]
    miss = 0
    guessed_letters = []

    while "_" in hidden:
        print("\n(Guess) Enter a letter in the word:", ''.join(hidden), ">")
        guess = input("Enter a letter: ").lower()

        if not guess or not guess[0].isalpha():
            print("Please enter a valid letter.")
            continue

        guess = guess[0]

        if guess in guessed_letters:
            print("You already guessed", guess)
            continue
        guessed_letters.append(guess)

        if guess in current_word[1:-1]:  # skip first and last
            for i in range(1, len(current_word) - 1):
                if current_word[i] == guess:
                    hidden[i] = guess
        else:
            print(guess, "is not in the word.")
            miss += 1

    print("\nğŸ‰ The word is:", ''.join(current_word))
    print("You missed", miss, "time(s).")

    again = input("\nDo you want to play again? (y/n): ").lower()
    if again != 'y':
        break
