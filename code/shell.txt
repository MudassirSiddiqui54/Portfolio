#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    char input[1024];
    char *args[100];
    pid_t pid;

    while (1) {
        printf("MyShell> ");
        fgets(input, sizeof(input), stdin);
        input[strcspn(input, "\n")] = 0; // remove newline

        // Tokenize input
        char *token = strtok(input, " ");
        int i = 0;
        while (token != NULL) {
            args[i++] = token;
            token = strtok(NULL, " ");
        }
        args[i] = NULL; // execvp expects NULL-terminated array

        // Exit condition
if (strcmp(args[0], "exit") == 0)
    break;

// Handle 'cd' manually
if (strcmp(args[0], "cd") == 0) {
    if (args[1] == NULL) {
        printf("cd: missing argument\n");
    } else {
        if (chdir(args[1]) != 0) {
            perror("cd");
        }
    }
    continue; // don't fork after 'cd'
}

        pid = fork();
        if (pid == 0) {
            // Child process
            execvp(args[0], args);
            perror("Command failed");
            exit(1);
        } else {
            // Parent process
            wait(NULL);
        }
    }

    return 0;
}
